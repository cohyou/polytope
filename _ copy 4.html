#sort Bool

#func true : Bool
#func false : Bool
#func not : Bool -> Bool
#func and : Bool * Bool -> Bool
#func or : Bool * Bool -> Bool

#rule a : Bool | or <a, false> = a
#rule a : Bool | and <a, true> = a
#rule a b : Bool | or <a, b> = or <b, a>
#rule a b : Bool | and <a, b> = and <b, a>
#rule a : Bool | or <a, not a> = true
#rule a : Bool | and <a, not a> = false
#rule a b c : Bool | or <a, and <b, c>> = and <or <a, b>, or <a, c>>
